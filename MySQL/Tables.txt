CREATE TABLE IF NOT EXISTS Users (
		UserID						int				AUTO_INCREMENT,
		FirstName					CHAR(100) CHARACTER SET utf8 COLLATE utf8_unicode_ci,
		LastName					CHAR(100) CHARACTER SET utf8 COLLATE utf8_unicode_ci,
		IdentityNumber				CHAR(20) CHARACTER SET utf8 COLLATE utf8_unicode_ci,
		UserName					CHAR(100) CHARACTER SET utf8 COLLATE utf8_unicode_ci,
		PasswordHash				BLOB			NOT NULL,
		PasswordSalt				BLOB			NOT NULL,
		PasswordLastChange			datetime		NULL,
		Active						tinyint			NOT NULL,
		InsertDate					timestamp		NOT NULL DEFAULT CURRENT_TIMESTAMP,
		UpdateDate					timestamp		NOT NULL DEFAULT CURRENT_TIMESTAMP,
		PRIMARY KEY (UserID)
	);
